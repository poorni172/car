import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
cars_df = pd.read_csv("book3.csv")
print(cars_df)
a=plt.scatter(cars_df['MPG'],cars_df['Wt'],market = '*')
print(a)
plt.xlabel("MPG")
plt.ylabel("Weight")
plt.title('Relationship between the weight of the car and MPG')
plt.show()
trans = cars_df['am'].value counts()
print(trans)
trans.plot.box()
plt.xlabel("Transmission")
plt.title('Bpx plot to check the Frequency distribution of Transmission type of cars')
plt.show()
cars_df['MPG'].plot(kind='box')
plt.xlabel('MPG')
plt.title('Box and whisker plot of MPG')
plt.show()
min_value = np.min(cars_df['MPG'])
q1=np.percentile(cars_df['MPG'],25)
median=np.median(cars_df['MPG'])
q3=np.percentile(cars_df['MPG'],75)
max_value=np.max(cars_df['MPG'])
print('Minimum value:',min_value)
print('Q1:',q1)
print('Median:',median)
print('Q3:',q3)
print('Maximum value:',max_value)
